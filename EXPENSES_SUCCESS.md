## 🎓 أمثلة سريعة

### مثال 1: راتب موظف

```javascript
{
  expenseType: "salary",
  employeeName: "أحمد محمد",
  employeePosition: "عامل تغذية",
  salaryMonth: "2025-10",
  baseSalary: 5000,
  bonuses: 1000,
  deductions: 500,
  netSalary: 5500,  // ✅ محسوب تلقائياً
  paymentMethod: "bank_transfer"
}

النتيجة: 5000 + 1000 - 500 = 5500 ج ✅
```

### مثال 2: شراء أعلاف

```javascript
{
  expenseType: "feed",
  category: "🌾 أعلاف",
  amount: 12500,
  quantity: 500,
  unitPrice: 25,
  description: "علف تسمين مركز - 500 كجم",
  paymentMethod: "cash"
}
```

### مثال 3: فاتورة كهرباء

```javascript
{
  expenseType: "utilities",

  description: "فاتورة كهرباء - أكتوبر 2025",
  referenceNumber: "ELEC-2025-10",

}
```
